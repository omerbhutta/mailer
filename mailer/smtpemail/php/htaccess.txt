#Redirect from error page
ErrorDocument 404 https://astra-management.ca/error-page.php
#Redirect from www to without www
Options +FollowSymLinks
RewriteEngine On
RewriteBase /
RewriteCond %{HTTP_HOST} ^www.astra-management.ca$ [NC]
RewriteRule ^(.*)$ https://astra-management.ca/$1 [R=301,L]
#Redirect from index php to root
RewriteCond %{THE_REQUEST} ^.*/index\.php
RewriteRule ^(.*)index.php$ /$1 [R=301,L]
#Redirect from index html to root
RewriteCond %{THE_REQUEST} ^.*/index\.html
RewriteRule ^(.*)index.html$ /$1 [R=301,L]
#Redirect from pages with slash in the end url - to witout slash
RewriteCond %{REQUEST_FILENAME} !-d
RewriteCond %{REQUEST_URI} ^(.+)/$
RewriteRule ^(.+)/$ /$1 [R=301,L]
#Redirect from many slashes before index php or other pages
RewriteCond %{THE_REQUEST} //
RewriteRule .* /$0 [R=301,L]
#Redirect vnutr stranici with .php na stranici bez .php
RewriteCond %{REQUEST_FILENAME} !-d
RewriteCond %{REQUEST_FILENAME}\.php -f
RewriteRule ^(.*)$ $1.php
